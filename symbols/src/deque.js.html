1 'use strict';
  2 
  3 /**
  4  * @fileOverview Defines the Deque prototype and its methods.
  5  */
  6 module.exports = Deque;
  7 
  8 /**
  9  * Deque constructor.
 10  *
 11  * @class
 12  *
 13  * An Deque object acts much like the C++ std::deque container. Provides operations to add/remove objects from the
 14  * start and the end of the containuer.
 15  */
 16 function Deque() {
 17     this._head = new Deque.Node();
 18     this._tail = new Deque.Node();
 19     this._head._next = this._tail;
 20     this._tail._prev = this._head;
 21     this._size = 0;
 22 }
 23 
 24 /**
 25  * Deque Node constructor.
 26  *
 27  * @class
 28  *
 29  * A Node object simple maintains links to a Deque double linked list.
 30  */
 31 Deque.Node = function(data) {
 32     this._data = data;
 33     this._prev = null;
 34     this._next = null;
 35 };
 36 
 37 /**
 38  * Deque prototype.
 39  *
 40  * Defines the methods available to a Deque instance.
 41  */
 42 Deque.prototype = {
 43 
 44     /**
 45      * Determine if the container has no elements.
 46      *
 47      * @return true if so
 48      */
 49     empty: function () {
 50         return this._head._next === this._tail;
 51     },
 52 
 53     /**
 54      * Obtain the number of elements held in the container.
 55      *
 56      * @return {Number} current container capacity
 57      */
 58     size: function () {
 59         return this._size;
 60     },
 61 
 62     /**
 63      * Add a new Node instance to the back of the queue.
 64      *
 65      * @param {Object} data value to add to the queue
 66      */
 67     pushBack: function (data) {
 68         var node = new Deque.Node(data);
 69         node._prev = this._tail._prev;
 70         node._prev._next = node;
 71         node._next = this._tail;
 72         this._tail._prev = node;
 73         ++this._size;
 74     },
 75 
 76     /**
 77      * Remove a Node instance from the back of the queue.
 78      *
 79      * @returns {Object} held data value
 80      */
 81     popBack: function () {
 82         if (this.empty()) {
 83             throw new Error("popBack called on empty deque");
 84         }
 85         --this._size;
 86         var node = this._tail._prev;
 87         this._tail._prev = node._prev;
 88         node._prev._next = this._tail;
 89         return node._data;
 90     },
 91 
 92     /**
 93      * Add a new Node instance to the front of the queue.
 94      *
 95      * @param {Object} data value to add to the queue
 96      */
 97     pushFront: function (data) {
 98         var node = new Deque.Node(data);
 99         node._next = this._head._next;
100         node._next._prev = node;
101         node._prev = this._head;
102         this._head._next = node;
103         ++this._size;
104     },
105 
106     /**
107      * Remove a Node instance from the front of the queue.
108      *
109      * @returns {Object} held data value
110      */
111     popFront: function () {
112         if (this.empty()) {
113             throw new Error("popFront called on empty deque");
114         }
115         --this._size;
116         var node = this._head._next;
117         this._head._next = node._next;
118         node._next._prev = this._head;
119         return node._data;
120     }
121 };
